5. src/sim/hemodynamics.rs — NEW: Multi-Phase Hemodynamics & Margination (Improvement #1)

Extends lbm.rs rather than replacing it: ParticulateField wraps an LbmField for the plasma phase and adds a second, coarser Eulerian field tracking local RBC volume fraction (hematocrit), from which a margination force is derived and applied to nanoparticles each tick. This reproduces the well-documented phenomenon where RBCs occupy the vessel core (the Fahraeus-Lindqvist effect) and push smaller particles into the cell-free layer near the wall — directly governing how often a nanobot actually contacts the endothelium versus staying entrained in the fast-moving core.

#![cfg(feature = "hemo")] use crate::sim::lbm::LbmField; use crate::sim::environment::Voxel; /// Per-voxel red-blood-cell volume fraction field. Modeled as an advected /// scalar (RBCs are large enough that their own diffusion is negligible on /// the timescale of a simulation tick) with a lift term that concentrates /// mass toward the vessel centerline — the inverse of what nanoparticles do. pub struct RbcPhase { pub dims: [usize; 3], pub fraction: Vec<f64>, // 0.0 (plasma only) .. ~0.6 (packed RBCs) } impl RbcPhase { pub fn new_uniform(dims: [usize; 3], hematocrit: f64) -> Self { let n = dims[0] * dims[1] * dims[2]; RbcPhase { dims, fraction: vec![hematocrit; n] } } #[inline] fn idx(&self, x: usize, y: usize, z: usize) -> usize { x + self.dims[0] * (y + self.dims[1] * z) } /// Advects RBC fraction downstream with the flow field and re-applies a /// radial "focusing" term biasing mass toward locally faster/central /// flow (Segre-Silberberg-style inertial focusing, coarse approximation). pub fn step(&mut self, lbm: &LbmField, wall_distance: &[f64]) { let (dx, dy, dz) = (self.dims[0], self.dims[1], self.dims[2]); let mut next = self.fraction.clone(); for x in 0..dx { for y in 0..dy { for z in 0..dz { let i = self.idx(x, y, z); if lbm.solid[i] { next[i] = 0.0; continue; } let (_rho, u) = lbm.macroscopic(i); let speed = (u[0] * u[0] + u[1] * u[1] + u[2] * u[2]).sqrt(); // Bias toward higher-speed (central) regions: RBCs migrate // away from the wall as local wall-distance increases their // effective centering force, capped to keep mass bounded. let focus_gain = 0.01 * speed * wall_distance[i].min(3.0); next[i] = (self.fraction[i] + focus_gain).clamp(0.0, 0.6); } } } self.fraction = next; } #[inline] pub fn at(&self, x: usize, y: usize, z: usize) -> f64 { self.fraction[self.idx(x, y, z)] } } /// Two-phase hemodynamics container: single-phase LBM field for bulk flow, /// plus an RBC phase field, plus derived per-voxel wall distance used both /// for RBC focusing and for the margination force applied to nanobots. pub struct ParticulateField { pub lbm: LbmField, pub rbc: RbcPhase, pub wall_distance: Vec<f64>, } impl ParticulateField { pub fn new(dims: [usize; 3], hematocrit: f64, inlet_velocity: [f64; 3]) -> Self { let lbm = LbmField::new_equilibrium(dims, 1.0, inlet_velocity); let rbc = RbcPhase::new_uniform(dims, hematocrit); let wall_distance = compute_wall_distance(&lbm); ParticulateField { lbm, rbc, wall_distance } } pub fn step(&mut self, tau: f64, inlet_velocity: [f64; 3]) { self.lbm.step(tau, inlet_velocity); self.rbc.step(&self.lbm, &self.wall_distance); } /// The core deliverable of Improvement #1: a lateral force vector /// pushing a nanoparticle at `pos` (grid coords) away from RBC-dense /// regions and toward the cell-free layer near the wall, scaled by /// `margination_coeff` and by particle size (smaller particles marginate /// more readily — captured via `size_factor` computed by the caller from /// `Params::size_nm`). pub fn margination_force(&self, x: usize, y: usize, z: usize, size_factor: f64, coeff: f64) -> [f64; 3] { let center_frac = self.rbc.at(x, y, z); let mut grad = [0.0; 3]; let (dx, dy, dz) = (self.rbc.dims[0], self.rbc.dims[1], self.rbc.dims[2]); let sample = |xi: isize, yi: isize, zi: isize| -> f64 { if xi < 0 || yi < 0 || zi < 0 || xi as usize >= dx || yi as usize >= dy || zi as usize >= dz { return center_frac; } self.rbc.at(xi as usize, yi as usize, zi as usize) }; let (xi, yi, zi) = (x as isize, y as isize, z as isize); grad[0] = sample(xi - 1, yi, zi) - sample(xi + 1, yi, zi); grad[1] = sample(xi, yi - 1, zi) - sample(xi, yi + 1, zi); grad[2] = sample(xi, yi, zi - 1) - sample(xi, yi, zi + 1); // Force points down the RBC-fraction gradient (away from crowding), // scaled by how "marginate-able" this particle size is. for a in 0..3 { grad[a] *= coeff * size_factor; } grad } /// Fraction of voxels near the wall that are "capture-eligible" this /// tick (low RBC fraction + within endothelial contact distance) — /// exposed for the `capture_efficiency` statistic (Improvement #1 goal /// of predicting how often a bot actually touches the target tissue). pub fn wall_contact_fraction(&self, contact_distance: f64) -> f64 { let n = self.wall_distance.len(); if n == 0 { return 0.0; } let eligible = self.wall_distance.iter().zip(self.rbc.fraction.iter()) .filter(|(&wd, &frac)| wd <= contact_distance && frac < 0.15) .count(); eligible as f64 / n as f64 } } /// Multi-source BFS-style distance-to-nearest-solid, in voxel units — /// cheap enough to run once (static anatomy) or on `bvh_rebuild_every`. fn compute_wall_distance(lbm: &LbmField) -> Vec<f64> { let (dx, dy, dz) = (lbm.dims[0], lbm.dims[1], lbm.dims[2]); let n = dx * dy * dz; let mut dist = vec![f64::INFINITY; n]; let mut frontier = Vec::new(); let idx = |x: usize, y: usize, z: usize| x + dx * (y + dy * z); for x in 0..dx { for y in 0..dy { for z in 0..dz { let i = idx(x, y, z); if lbm.solid[i] { dist[i] = 0.0; frontier.push((x, y, z)); } }}} while let Some((x, y, z)) = frontier.pop() { let d0 = dist[idx(x, y, z)]; for (nx, ny, nz) in [ (x.wrapping_sub(1), y, z), (x + 1, y, z), (x, y.wrapping_sub(1), z), (x, y + 1, z), (x, y, z.wrapping_sub(1)), (x, y, z + 1), ] { if nx >= dx || ny >= dy || nz >= dz { continue; } let j = idx(nx, ny, nz); if d0 + 1.0 < dist[j] { dist[j] = d0 + 1.0; frontier.push((nx, ny, nz)); } } } dist } /// Statistics surfaced to `stats::TickRecord` for dosage-prediction analysis. #[derive(Clone, Copy, Debug, Default)] pub struct MarginationStats { pub mean_wall_distance_marginated: f64, pub wall_contact_fraction: f64, } 

Wiring into environment.rs / engine.rs: Environment gains pub particulate: Option<ParticulateField>, populated by enable_particulate_hemo() when Params::hemo_model == HemoModel::Particulate (mirroring how V2's enable_lbm worked). Environment::diffuse_and_decay dispatches to ParticulateField::step instead of LbmField::step in that mode, and Simulation::step's SDE integration pass adds margination_force(...) as one more additive term alongside rl_thrust, mag_force, and acoustic_force.

6. src/sim/gnn.rs — NEW: Swarm Intelligence via Graph Neural Networks (Improvement #2)

Nanobots are nodes in a dynamic k-NN graph rebuilt every tick from Environment spatial queries. A lightweight message-passing layer (two rounds of "gather neighbor state → MLP → update own state") lets the swarm develop emergent behaviors — cooperative docking (bots link into a physical plug to stem a hemorrhage) and swarm-steering (front-of-cluster bots broadcast pressure changes backward) — without requiring a full deep-learning framework; weights are small dense matrices loaded from a .npz file trained offline (see §9, PyO3 bridge).

#![cfg(feature = "gnn")] use ndarray::{Array1, Array2}; use petgraph::graph::UnGraph; use crate::sim::agent::NanoBCIAgent; /// Per-agent feature vector fed into the GNN: physical + biological state, /// deliberately overlapping with `RlObservation` so the two policies are /// interchangeable behind `Params::swarm_backend` / `rl_enabled`. #[derive(Clone, Debug)] pub struct SwarmMessage { pub node_id: usize, pub pos: [f64; 3], pub vel: [f64; 3], pub local_pressure: f64, pub docked: bool, pub cargo_amount: f64, } /// The dynamic neighbor graph, rebuilt each tick from agent positions /// within `Params::swarm_graph_radius`, capped at `swarm_max_degree` /// nearest neighbors per node to bound the O(n*k) message-passing cost. pub struct SwarmGraph { pub graph: UnGraph<usize, f64>, // edge weight = inter-agent distance } impl SwarmGraph { pub fn build(agents: &[NanoBCIAgent], radius: f64, max_degree: usize) -> Self { let mut graph = UnGraph::<usize, f64>::new_undirected(); let node_indices: Vec<_> = (0..agents.len()).map(|i| graph.add_node(i)).collect(); for i in 0..agents.len() { let mut neighbors: Vec<(usize, f64)> = (0..agents.len()) .filter(|&j| j != i) .map(|j| (j, dist(agents[i].pos, agents[j].pos))) .filter(|&(_, d)| d < radius) .collect(); 3], f64)> { let mut out = Vec::with_capacity(messages.len()); for node_idx in graph.graph.node_indices() { let i = graph.graph[node_idx]; let neighbor_feats: Vec<Array1<f64>> = graph.graph.neighbors(node_idx) .map(|n| Self::feature_vec(&messages[graph.graph[n]])) .collect(); let cooperate = action.get(3).copied().unwrap_or(0.0).clamp(0.0, 1.0); out.push((i, thrust, cooperate)); } out } } fn dist(a: [f64; 3], b: [f64; 3]) -> f64 { ((a[0]-b[0]).powi(2) + (a[1]-b[1]).powi(2) + (a[2]-b[2]).powi(2)).sqrt() } 

Cooperative docking (hemorrhage plug): when Params::enable_cooperative_docking is set and a node's cooperate_strength exceeds a threshold while docked, Simulation::step links it into a HemorrhagePlug group (a Vec<usize> of agent ids sharing a target voxel); the environment treats a fully-formed plug's voxel as inside_geometry = false for the LBM/particulate solver, mechanically stopping flow through it — a direct, testable implementation of "bots link together to create a physical barrier."

7. src/sim/kinetics.rs — NEW: High-Fidelity Ligand-Receptor Binding Kinetics (Improvement #5)

Replaces the flat endocytosis_prob roll with real binding kinetics: a bot's surface ligand density and the target cell's receptor density combine with literature/lab-sourced Ka/Kd values to compute an instantaneous binding probability per tick, so different synthetic coatings can be compared for "capture efficiency" the way the improvements doc describes.

#![cfg(feature = "kinetics")] use serde::{Deserialize, Serialize}; /// Surface ligand properties for one nanoparticle "coating" design. #[derive(Clone, Copy, Serialize, Deserialize, Debug)] pub struct LigandProfile { pub density_per_um2: f64, // ligands / um^2 on the particle surface pub ka: f64, // association rate constant pub kd_off: f64, // dissociation rate constant } /// Target-cell receptor properties (varies by tissue/tumor subtype). #[derive(Clone, Copy, Serialize, Deserialize, Debug)] pub struct ReceptorProfile { pub density_per_um2: f64, } /// Encapsulates the law-of-mass-action binding model used each tick a /// docked-eligible agent is in contact with tissue. pub struct BindingKinetics; impl BindingKinetics { /// Equilibrium dissociation constant, Kd = k_off / k_on — smaller Kd /// means tighter (higher-affinity) binding. pub fn kd(ligand: &LigandProfile) -> f64 { if ligand.ka <= 0.0 { return f64::INFINITY; } ligand.kd_off / ligand.ka } /// Fractional receptor occupancy at equilibrium under the standard /// single-site binding isotherm, given local effective ligand /// "concentration" derived from surface density and contact area — /// this is the "virtual keys fit virtual locks" mechanism from the /// improvements doc, made quantitative. pub fn occupancy(ligand: &LigandProfile, receptor: &ReceptorProfile, shear_stress: f64) -> f64 { let kd = Self::kd(ligand); if kd.is_infinite() { return 0.0; } let effective_ligand = ligand.density_per_um2; let raw = effective_ligand / (kd + effective_ligand); // Shear stress strips marginally-bound ligand-receptor pairs before // enough bonds accumulate to resist flow — high-flow vessels need // higher avidity (more ligands / lower Kd) to achieve the same // capture efficiency, matching the improvement's stated goal of // testing coatings under "high-flow environments." let shear_penalty = (-0.02 * shear_stress).exp(); (raw * shear_penalty * (receptor.density_per_um2 / 150.0).min(2.0)).clamp(0.0, 1.0) } /// Per-tick stochastic docking decision replacing `endocytosis_prob`. pub fn should_dock(ligand: &LigandProfile, receptor: &ReceptorProfile, shear_stress: f64) -> bool { rand::random::<f64>() < Self::occupancy(ligand, receptor, shear_stress) } /// "Capture efficiency" metric for a coating design: mean occupancy /// across a batch of (shear_stress) samples along a vessel — this is /// exactly the comparison the improvements doc asks for ("test /// different synthetic coatings ... to see which has the highest /// capture efficiency"). pub fn capture_efficiency(ligand: &LigandProfile, receptor: &ReceptorProfile, shear_samples: &[f64]) -> f64 { if shear_samples.is_empty() { return 0.0; } shear_samples.iter().map(|&s| Self::occupancy(ligand, receptor, s)).sum::<f64>() / shear_samples.len() as f64 } } 

Wiring: NanoBCIAgent gains pub ligand: LigandProfile (built from Params::ligand_ka/ligand_kd_off/ligand_density_surface); Environment::Voxel (or, in hemo mode, ParticulateField) supplies shear_stress via LbmField::shear_stress_at. In agent::step, the AgentState::Brain -> AgentState::Docked transition dispatches on Params::docking_model: Probabilistic keeps the V1/V2 endocytosis_prob roll; LigandReceptor calls BindingKinetics::should_dock instead.

8. src/sim/bvh.rs — NEW: BVH-Accelerated Anatomy & Collision (Improvement #6)

Replaces mesh_import.rs's O(n) ray-parity / nearest-triangle scan (fine for a coarse voxel grid, prohibitive for whole-organ, MRI-resolution meshes) with a proper Bounding Volume Hierarchy built on parry3d-f64, the same class of structure used in ray-tracing engines. AnatomyBvh answers both point-containment (inside/outside anatomy, for SDF rasterization) and nearest-surface-distance queries in O(log n) instead of O(n).

#![cfg(feature = "bvh")] use anyhow::Result; use parry3d_f64::bounding_volume::Aabb; use parry3d_f64::math::{Point, Isometry}; use parry3d_f64::query::{PointQuery, Ray, RayCast}; use parry3d_f64::shape::TriMesh; 10. src/sim/calibration.rs — NEW: Automated Calibration via Evolutionary Strategies (Improvement #7)

Implements "Auto-Tune": given real lab observations (e.g. "we injected N particles and X% reached the liver by tick T"), a (μ, λ) evolution-strategy search back-calculates unknown biophysical variables — opsonization rate, zeta potential, endocytosis probability — that make the simulation reproduce the lab data. This generalizes optimizer::grid_search (V1/V2, exhaustive and only practical over 2-3 parameters) to a much higher-dimensional search.

#![cfg(feature = "autotune")] use rand::Rng; use rayon::prelude::*; use serde::{Deserialize, Serialize}; use crate::sim::params::{Params, Scenario}; use crate::sim::engine::Simulation; /// One real-world measurement to fit against, e.g. "5% of injected dose /// reached the liver region by tick 3000." #[derive(Clone, Deserialize, Serialize, Debug)] pub struct LabObservation { pub tick: u64, pub target_region_center_nm: [f64; 11. src/gpu/resident.rs — NEW: Full GPU-Resident Simulation (Improvement #4)

V2's gpu::GpuContext uploads agent state, dispatches one kernel, and reads results back every tick — the "bus bottleneck" the improvements doc calls out. resident.rs keeps the LBM/particulate field, agent positions/velocities, and Behavior Tree evaluation in persistent GPU storage buffers across the entire run; the CPU only issues kernel dispatches and reads back a small aggregated stats buffer periodically (for stats::TickRecord), not full agent state.

dispatch(&mut encoder, &self.integrate_pipeline, self.n_agents, &self.device, &self.agents_buffer, &ubuf); encoder.copy_buffer_to_buffer(&self.stats_buffer, 0, &self.stats_staging, 0, 256); self.queue.submit(Some(encoder.finish())); } // src/gpu/shaders/behavior_tree.wgsl — GPU-side Conditional Toxicity check, // mirroring sim/behavior_tree.rs::default_conditional_toxicity_tree without // a CPU round-trip. 18. Feature Flags & Build Profiles (V3)

# Fast iteration / CI — BVH + auto-tune are cheap pure-Rust, on by default cargo build --release # Full clinical-grade build — every V3 subsystem cargo build --release --features full # Research subset: particulate hemodynamics + ligand-receptor kinetics, # no GPU/Python/GNN cargo build --release --features "hemo,kinetics" # GPU-resident billion-agent runs cargo build --release --features gpu-resident # Build the Python wheel (from mitonano-py/) cd mitonano-py && maturin develop --release --features python,full 

HemoModel defaults to SinglePhase, SwarmBackend to None, and DockingModel to Probabilistic, so a plain cargo build --release run stays behaviorally identical to V2. bvh and autotune ship in default because they're pure-Rust, dependency-light, and strictly improve correctness/capability with no opt-in cost — every other V3 subsystem remains explicit via Params fields or CLI flags, preserving V2's "additive, feature-gated" philosophy.

19. Suggested Next Steps (V4 candidates)

Port the fixed 3-check behavior_tree.wgsl kernel into a data-driven GPU BT interpreter (upload the BTNode tree as a flat instruction buffer) so GPU-resident runs aren't limited to the hardcoded default tree.

Extend hemodynamics.rs's RBC phase from a scalar volume-fraction field to a full immersed-boundary RBC membrane model (discrete deformable capsules) for regimes where individual-cell shape (not just bulk hematocrit) matters, e.g. capillary-diameter vessels.

Train GnnPolicy weights end-to-end via the PyO3 bridge (PPO/MAPPO against a headless Simulation loop) rather than hand-authored MLP weights, closing the loop the improvements doc describes for Improvement #2.

Extend AutoTuner from a (μ, λ) evolution strategy to CMA-ES proper (adaptive covariance matrix) once the tunable-parameter count grows past what isotropic Gaussian mutation handles well.

Wire AnatomyBvh into hemodynamics.rs's wall-distance computation directly (currently a separate BFS in compute_wall_distance), so a single BVH serves mesh import, agent collision, and margination geometry.

Add a numpy zero-copy path for PySimulation.positions() on GPU-resident runs (currently requires a CPU-side agent snapshot; a resident run would need one explicit readback dispatch first).
