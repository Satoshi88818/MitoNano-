MitoNano Engine — Integrated Codebase

This document merges the original MitoNano BCI nanoparticle-delivery architecture with the Multi-Scale Biological Engineering improvements (voxel environment, agent logic gates, external actuation, stigmergy/swarm signaling, and expanded pharmacokinetic parameters). It is a single, coherent Rust workspace you can drop into a project.

5. src/sim/environment.rs

New module implementing the voxel map and pheromone / stigmergy layer from the improvements doc.

use serde::{Deserialize, Serialize}; #[derive(Clone, Copy, Serialize, Deserialize, Debug, Default)] pub struct Voxel { pub ph: f64, pub oxygen_level: f64, pub chemical_gradient: f64, pub flow_vector: [f64; 3], pub drug_concentration: f64, pub pheromone: f64, /// 1.0 = healthy tissue, 0.0 = fully diseased/damaged (tumor, wound, clot) pub tissue_health: f64, } impl Voxel { pub fn healthy() -> Self { Voxel { ph: 7.4, oxygen_level: 0.9, chemical_gradient: 0.0, flow_vector: [0.0, 0.0, 0.0], drug_concentration: 0.0, pheromone: 0.0, tissue_health: 1.0, } } pub fn diseased(ph: f64, oxygen: f64) -> Self { Voxel { ph, oxygen_level: oxygen, chemical_gradient: 0.0, flow_vector: [0.0, 0.0, 0.0], drug_concentration: 0.0, pheromone: 0.0, tissue_health: 0.15, } } } /// External fields for magnetic / acoustic actuation, applied uniformly /// (a future extension could make these spatially varying too). #[derive(Clone, Copy, Serialize, Deserialize, Debug, Default)] pub struct ExternalFields { pub magnetic_gradient: [f64; 3], // Tesla per meter pub acoustic_pressure: f64, // Pascal } pub struct Environment { pub grid: Vec<Voxel>, pub dimensions: [usize; 3], pub resolution_nm: f64, } impl Environment { pub fn new_uniform(dimensions: [usize; 3], resolution_nm: f64) -> Self { let n = dimensions[0] * dimensions[1] * dimensions[2]; Environment { grid: vec![Voxel::healthy(); n], dimensions, resolution_nm, } } #[inline] fn idx(&self, x: usize, y: usize, z: usize) -> usize { x + self.dimensions[0] * (y + self.dimensions[1] * z) } /// Converts a continuous nm position into a clamped voxel coordinate. fn pos_to_voxel(&self, pos: [f64; 3]) -> (usize, usize, usize) { let clamp = |v: f64, max: usize| -> usize { let idx = (v / self.resolution_nm).floor().max(0.0) as usize; idx.min(max.saturating_sub(1)) }; ( clamp(pos[0], self.dimensions[0]), clamp(pos[1], self.dimensions[1]), clamp(pos[2], self.dimensions[2]), ) } pub fn voxel_at(&self, pos: [f64; 3]) -> &Voxel { let (x, y, z) = self.pos_to_voxel(pos); &self.grid[self.idx(x, y, z)] } pub fn voxel_at_mut(&mut self, pos: [f64; 3]) -> &mut Voxel { let (x, y, z) = self.pos_to_voxel(pos); let i = self.idx(x, y, z); &mut self.grid[i] } /// "Paint" a spherical diseased region (tumor / wound / clot) into the /// grid — the digital-twin interaction described in the improvements doc. pub fn paint_region(&mut self, center_nm: [f64; 3], radius_nm: f64, ph: f64, oxygen: f64) { let (cx, cy, cz) = self.pos_to_voxel(center_nm); let r_vox = (radius_nm / self.resolution_nm).ceil() as isize; let (dx, dy, dz) = (self.dimensions[0] as isize, self.dimensions[1] as isize, self.dimensions[2] as isize); for x in (cx as isize - r_vox).max(0)..(cx as isize + r_vox).min(dx) { for y in (cy as isize - r_vox).max(0)..(cy as isize + r_vox).min(dy) { for z in (cz as isize - r_vox).max(0)..(cz as isize + r_vox).min(dz) { let d2 = ((x - cx as isize).pow(2) + (y - cy as isize).pow(2) + (z - cz as isize).pow(2)) as f64; if d2 <= (r_vox * r_vox) as f64 { let i = self.idx(x as usize, y as usize, z as usize); self.grid[i] = Voxel::diseased(ph, oxygen); } } } } } /// Gradient of pheromone concentration at a position, used for /// chemotaxis (stigmergy). Returns a unit-ish direction vector. pub fn get_gradient(&self, pos: [f64; 3]) -> [f64; 3] { let (x, y, z) = self.pos_to_voxel(pos); let (dx_max, dy_max, dz_max) = (self.dimensions[0], self.dimensions[1], self.dimensions[2]); let sample = |xi: isize, yi: isize, zi: isize| -> f64 { if xi < 0 || yi < 0 || zi < 0 || xi as usize >= dx_max || yi as usize >= dy_max || zi as usize >= dz_max { return 0.0; } self.grid[self.idx(xi as usize, yi as usize, zi as usize)].pheromone }; let (xi, yi, zi) = (x as isize, y as isize, z as isize); let gx = sample(xi + 1, yi, zi) - sample(xi - 1, yi, zi); let gy = sample(xi, yi + 1, zi) - sample(xi, yi - 1, zi); let gz = sample(xi, yi, zi + 1) - sample(xi, yi, zi - 1); [gx, gy, gz] } pub fn deposit_pheromone(&mut self, pos: [f64; 3], amount: f64) { let v = self.voxel_at_mut(pos); v.pheromone += amount; } /// Diffuse drug concentration and decay pheromones each tick. Kept /// simple (no full PDE solve) — a cheap box-blur pass is enough for /// simulation purposes and stays fast at grid sizes like 32^3-64^3. pub fn diffuse_and_decay(&mut self, diffusion_coeff: f64, pheromone_decay: f64) { let n = self.grid.len(); let mut new_drug = vec![0.0; n]; let (dx, dy, dz) = (self.dimensions[0], self.dimensions[1], self.dimensions[2]); for x in 0..dx { for y in 0..dy { for z in 0..dz { let i = self.idx(x, y, z); let mut sum = self.grid[i].drug_concentration; let mut count = 1.0; let neighbors = [ (x.wrapping_sub(1), y, z), (x + 1, y, z), (x, y.wrapping_sub(1), z), (x, y + 1, z), (x, y, z.wrapping_sub(1)), (x, y, z + 1), ]; for (nx, ny, nz) in neighbors { if nx < dx && ny < dy && nz < dz { sum += self.grid[self.idx(nx, ny, nz)].drug_concentration; count += 1.0; } } let avg = sum / count; new_drug[i] = self.grid[i].drug_concentration + diffusion_coeff * (avg - self.grid[i].drug_concentration); } } } for (i, v) in self.grid.iter_mut().enumerate() { v.drug_concentration = new_drug[i].max(0.0); v.pheromone = (v.pheromone * (1.0 - pheromone_decay)).max(0.0); } } }

use serde::{Deserialize, Serialize}; use crate::sim::environment::Voxel; use crate::sim::params::Params; #[derive(Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Debug)] pub enum AgentState { Blood, Brain, Docked, // attached to a target cell/tissue, ready to act Cleared, } #[derive(Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Debug)] pub enum LogicState { Ready, Released, Depleted, } #[derive(Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Debug)] pub enum CargoType { None, ChemoDrug, GrowthFactor, Antitoxin, } /// The decision-making component described in the improvements doc: /// a simple AND-gate over local environment conditions plus a target /// protein signature (e.g. a tumor-surface marker). #[derive(Clone, Copy, Serialize, Deserialize, Debug)] pub struct AgentLogic { pub activation_threshold_ph: f64, pub target_protein_signature: u64, pub state: LogicState, pub cargo: CargoType, pub cargo_amount: f64, } impl AgentLogic { pub fn new(cargo: CargoType, activation_threshold_ph: f64, target_protein_signature: u64) -> Self { AgentLogic { activation_threshold_ph, target_protein_signature, state: LogicState::Ready, cargo, cargo_amount: 1.0, } } /// AND-gate: local pH below threshold AND agent is docked at a /// matching target. 3], pub vel: [f64; 3], pub magnetic_susceptibility: f64, // Pharmacokinetic state pub protein_corona: f64, // 0 (clean) .. 1 (fully opsonized) // Decision logic pub logic: AgentLogic, } impl NanoBCIAgent { pub fn new(id: usize, params: &Params, pos: [f64; 3]) -> Self { let cargo = match params.scenario { crate::sim::params::Scenario::Oncology => CargoType::ChemoDrug, crate::sim::params::Scenario::RegenerativeMedicine => CargoType::GrowthFactor, crate::sim::params::Scenario::Detoxification => CargoType::Antitoxin, crate::sim::params::Scenario::Bci => CargoType::None, }; NanoBCIAgent { id, state: AgentState::Blood, energy: 1.0, age: 0, lifetime: params.lifetime, signal_quality: 0.0, net_signal: 0.0, is_off_target: false, pos, vel: [0.0, 0.0, 0.0], magnetic_susceptibility: params.np_type.magnetic_susceptibility_base(), protein_corona: 0.0, logic: AgentLogic::new(cargo, params.activation_threshold_ph, params.target_protein_signature), } } /// Per-tick biological update: aging, opsonization, clearance, and /// state transitions (Blood -> Brain/Docked -> Cleared). pub fn step(&mut self, brain_agents: &[&NanoBCIAgent], params: &Params, voxel: &Voxel) { self.age += 1; // Protein-corona accumulation raises effective clearance and // reduces ligand-binding effectiveness over time. self.protein_corona = (self.protein_corona + params.protein_corona_rate).min(1.0); // Biodegradation via half-life (simple exponential decay model). let decay_prob = 1.0 - (-0.693_f64 / params.half_life).exp(); if rand::random::<f64>() < decay_prob || self.age as f64 > params.lifetime as f64 { self.state = AgentState::Cleared; return; } match self.state { AgentState::Blood => { let effective_bbb = params.bbb_prob * (1.0 - self.protein_corona * 0.5) * (1.0 + params.ligand_density * 0.3); if rand::random::<f64>() < effective_bbb { self.state = AgentState::Brain; } let clearance = params.np_type.clearance_base() * (1.0 + self.protein_corona); if rand::random::<f64>() < clearance { self.state = AgentState::Cleared; } } AgentState::Brain => { self.signal_quality = params.efficiency * (1.0 - self.protein_corona * 0.3); if params.agent_network && !brain_agents.is_empty() { let nearby: Vec<&&NanoBCIAgent> = brain_agents .iter() .filter(|a| a.id != self.id && dist(a.pos, self.pos) < params.net_radius) .collect(); } } else { self.net_signal = self.signal_quality; } } else { self.net_signal = self.signal_quality; } // Move toward "docked" if endocytosis succeeds at target. if voxel.tissue_health < 0.3 && rand::random::<f64>() < params.endocytosis_prob { self.state = AgentState::Docked; } } AgentState::Docked => { self.process_logic(voxel); } AgentState::Cleared => {} } } /// The "smart bot" decision function from the improvements doc: /// if local conditions AND target signature match, release cargo. pub fn process_logic(&mut self, voxel: &Voxel) { let matched_target = self.logic.target_protein_signature == 0 || !self.is_off_target; if self.logic.should_release(voxel, matched_target) { self.release_cargo(); } } fn release_cargo(&mut self) { self.logic.state = LogicState::Released; self.logic.cargo_amount = 0.0; } pub fn is_at_target(&self) -> bool { self.state == AgentState::Docked } } fn dist(a: [f64; 7. src/sim/sde.rs

Original Wiener-process integrator, extended with the external magnetic/acoustic force integration step.

use rand::Rng; use rand_distr::{Distribution, Normal}; use crate::sim::agent::NanoBCIAgent; use crate::sim::environment::ExternalFields; /// Single-axis Wiener increment: dW ~ N(0, sigma^2 * dt) pub fn wiener_step(sigma: f64, dt: f64, rng: &mut impl Rng) -> f64 { let normal = Normal::new(0.0, sigma * dt.sqrt()).unwrap(); normal.sample(rng) } /// Magnetic force on a particle: F = susceptibility * gradient (simplified /// linear model — swap in the full dipole equation if you need SI-accurate /// magnitudes). fn calculate_mag_force(susceptibility: f64, gradient: [f64; 3]) -> [f64; 3] { [ susceptibility * gradient[0], susceptibility * gradient[1], susceptibility * gradient[2], ] } /// Acoustic radiation force approximated as an isotropic push along the /// current velocity heading (a real acoustic-tweezer model would use a /// standing-wave pressure field gradient instead). fn calculate_acoustic_force(pressure: f64, heading: [f64; 3]) -> [f64; 3] { let mag = (heading[0].powi(2) + heading[1].powi(2) + heading[2].powi(2)).sqrt(); if mag < 1e-9 { return [0.0, 0.0, 0.0]; } let scale = pressure * 1e-6; // scaling factor into simulation units [heading[0] / mag * scale, heading[1] / mag * scale, heading[2] / mag * scale] } /// Integrates one step of Brownian motion plus external actuation plus /// chemotactic steering (pheromone gradient), and advances position. pub fn integrate_step( agent: &mut NanoBCIAgent, sigma: f64, dt: f64, fields: &ExternalFields, chemotaxis_gradient: [f64; 3], rng: &mut impl Rng, ) { let mag_force = calculate_mag_force(agent.magnetic_susceptibility, fields.magnetic_gradient); let acoustic_force = calculate_acoustic_force(fields.acoustic_pressure, agent.vel); for axis in 0..3 { let brownian = wiener_step(sigma, dt, rng); agent.vel[axis] += (mag_force[axis] + acoustic_force[axis]) * dt + chemotaxis_gradient[axis] * 0.05 + brownian; // mild damping so velocities don't diverge agent.vel[axis] *= 0.98; agent.pos[axis] += agent.vel[axis] * dt; } }

use rayon::prelude::*; use rand::thread_rng; use crate::sim::agent::{AgentState, NanoBCIAgent}; use crate::sim::environment::{Environment, ExternalFields}; use crate::sim::params::Params; use crate::sim::sde::integrate_step; use crate::sim::stats::History; pub struct Simulation { pub params: Params, pub agents: Vec<NanoBCIAgent>, pub env: Environment, pub fields: ExternalFields, pub tick: u64, pub history: History, } impl Simulation { pub fn new(params: Params) -> Self { let env = Environment::new_uniform(params.grid_dims, params.voxel_resolution_nm); let fields = ExternalFields { magnetic_gradient: params.magnetic_gradient, acoustic_pressure: params.acoustic_pressure, }; let mut agents = Vec::with_capacity(params.num_agents); let center = [ (params.grid_dims[0] as f64 * params.voxel_resolution_nm) / 2.0, (params.grid_dims[1] as f64 * params.voxel_resolution_nm) / 2.0, (params.grid_dims[2] as f64 * params.voxel_resolution_nm) / 2.0, ]; for id in 0..params.num_agents { agents.push(NanoBCIAgent::new(id, &params, center)); } Simulation { params, agents, env, fields, tick: 0, history: History::new(), } } /// Digital-twin interaction: paint a diseased region (tumor, wound, /// clot) into the environment before or during a run. pub fn paint_region(&mut self, center_nm: [f64; 3], radius_nm: f64, ph: f64, oxygen: f64) { self.env.paint_region(center_nm, radius_nm, ph, oxygen); } pub fn step(&mut self) { self.tick += 1; let dt = 1.0; let brain_agents: Vec<NanoBCIAgent> = self .agents .iter() .filter(|a| a.state == AgentState::Brain) .cloned() .collect(); let brain_refs: Vec<&NanoBCIAgent> = brain_agents.iter().collect(); // Biological / logic update (parallel), needs read-only env access. let env_ref = &self.env; let params_ref = &self.params; self.agents.par_iter_mut().for_each(|agent| { let voxel = env_ref.voxel_at(agent.pos); agent.step(&brain_refs, params_ref, voxel); }); // Physical movement + chemotaxis + external actuation (parallel). let fields = self.fields; let sigma = self.params.sde_sigma; let env_ref2 = &self.env; self.agents.par_iter_mut().for_each(|agent| { let mut rng = thread_rng(); let gradient = env_ref2.get_gradient(agent.pos); integrate_step(agent, sigma, dt, &fields, gradient, &mut rng); }); // Stigmergy: deposit pheromones + drug release at docked/target agents. // Sequential pass since it mutates the shared environment grid. for agent in self.agents.iter() { if agent.is_at_target() { self.env.deposit_pheromone(agent.pos, self.params.pheromone_deposit); } if agent.logic.state == crate::sim::agent::LogicState::Released { let v = self.env.voxel_at_mut(agent.pos); v.drug_concentration += 1.0; } } self.env .diffuse_and_decay(self.params.diffusion_coeff, self.params.pheromone_decay); // Drop cleared agents (swap-remove-friendly via retain). self.agents.retain(|a| a.state != AgentState::Cleared); self.history.record(self.tick, &self.agents, &self.env); } pub fn run(&mut self, max_ticks: u64) { for _ in 0..max_ticks { if self.agents.is_empty() { break; } self.step(); } } }

Extended to track pharmacokinetic (concentration-time) curves rather than just population counts, as called out in the improvements doc's "From Stats to Kinetics" summary point.

use serde::{Deserialize, Serialize}; use crate::sim::agent::{AgentState, NanoBCIAgent}; use crate::sim::environment::Environment; #[derive(Clone, Serialize, Deserialize, Debug, Default)] pub struct TickRecord { pub tick: u64, pub n_blood: usize, pub n_brain: usize, pub n_docked: usize, pub n_cleared_cum: usize, pub mean_signal_quality: f64, pub mean_net_signal: f64, pub mean_protein_corona: f64, /// Total drug/growth-factor/antitoxin concentration summed over the grid /// — the basis for a concentration-time (C-t) pharmacokinetic curve. pub total_drug_concentration: f64, pub max_pheromone: f64, } #[derive(Clone, Serialize, Deserialize, Debug, Default)] pub struct History { pub records: Vec<TickRecord>, cleared_cum: usize, } impl History { pub fn new() -> Self { History { records: Vec::new(), cleared_cum: 0 } } pub fn record(&mut self, tick: u64, agents: &[NanoBCIAgent], env: &Environment) { let n_blood = agents.iter().filter(|a| a.state == AgentState::Blood).count(); let n_brain = agents.iter().filter(|a| a.state == AgentState::Brain).count(); let n_docked = agents.iter().filter(|a| a.state == AgentState::Docked).count(); let mean = |f: fn(&NanoBCIAgent) -> f64| -> f64 { if agents.is_empty() { 0.0 } else { agents.iter().map(f).sum::<f64>() / agents.len() as f64 } }; let total_drug: f64 = env.grid.iter().map(|v| v.drug_concentration).sum(); let max_pher = env.grid.iter().map(|v| v.pheromone).fold(0.0, f64::max); self.records.push(TickRecord { tick, n_blood, n_brain, n_docked, n_cleared_cum: self.cleared_cum, mean_signal_quality: mean(|a| a.signal_quality), mean_net_signal: mean(|a| a.net_signal), mean_protein_corona: mean(|a| a.protein_corona), total_drug_concentration: total_drug, max_pheromone: max_pher, }); } pub fn latest(&self) -> Option<&TickRecord> { self.records.last() } } 

10. src/sim/optimizer.rs

Kept close to the original recommendation (grid / evolutionary / Bayesian-style search), generalized to optimize over the new PK/logic parameters too.

use rayon::prelude::*; use crate::sim::engine::Simulation; use crate::sim::params::Params; pub type ObjectiveFn = fn(&Params) -> f64; /// Simple parallel grid search over a Vec of candidate parameter sets. /// Swap in `argmin` (CMA-ES / differential evolution) or a Gaussian-process /// surrogate for smarter search once you need >~10^4 evaluations. pub fn grid_search(candidates: Vec<Params>, ticks: u64, objective: ObjectiveFn) -> (Params, f64) { candidates .into_par_iter() .map(|p| { let mut sim = Simulation::new(p.clone()); sim.run(ticks); let score = objective(&p); (p, score) }) .reduce( || (Params::default(), f64::NEG_INFINITY), |a, b| if a.1 >= b.1 { a } else { b }, ) } /// A minimal (mu, lambda) evolutionary strategy over a mutation closure, /// useful for tuning logic thresholds / cargo kinetics without pulling in /// a full optimization crate. pub fn evolve<F: Fn(&Params) -> Params + Sync>( seed: Params, generations: usize, population: usize, ticks: u64, mutate: F, objective: ObjectiveFn, ) -> (Params, f64) { let mut best = seed.clone(); let mut best_score = f64::NEG_INFINITY; for _ in 0..generations { let candidates: Vec<Params> = (0..population).map(|_| mutate(&best)).collect(); let (gen_best, gen_score) = grid_search(candidates, ticks, objective); if gen_score > best_score { best = gen_best; best_score = gen_score; } } (best, best_score) }

15. Example: Oncology Tumor-Targeting Sweep

use mitobci_engine::sim::params::{Params, Scenario}; use mitobci_engine::sim::engine::Simulation; use mitobci_engine::sim::optimizer::grid_search; fn tumor_kill_objective(params: &Params) -> f64 { let mut sim = Simulation::new(params.clone()); sim.paint_region([8000.0, 8000.0, 8000.0], 2500.0, 6.2, 0.15); sim.run(3000); // Reward runs that build up high local drug concentration quickly. sim.history.latest().map(|r| r.total_drug_concentration).unwrap_or(0.0) } fn main() { let mut candidates = Vec::new(); for ph_threshold in [6.0, 6.3, 6.5, 6.8] { for endocytosis in [0.05, 0.10, 0.15] { let mut p = Params::preset(Scenario::Oncology); p.activation_threshold_ph = ph_threshold; p.endocytosis_prob = endocytosis; candidates.push(p); } } let (best, score) = grid_search(candidates, 3000, tumor_kill_objective); println!( "Best: pH_threshold={} endocytosis={} -> score={:.3}", best.activation_threshold_ph, best.endocytosis_prob, score ); }

17. Suggested Next Steps

Replace the box-blur diffusion in Environment::diffuse_and_decay with a proper finite-difference or spectral solver if you need quantitatively accurate PK curves.

Swap optimizer::grid_search/evolve for argmin's CMA-ES once the parameter space grows past a handful of dimensions.

Extend AgentLogic from a single AND-gate to a small boolean-expression tree (AND/OR/NOT combinators) if you need multi-condition release logic (e.g. pH AND protein-match AND NOT healthy-tissue).

For the WASM front-end, stream FrameSnapshot plus a downsampled voxel slice (e.g. a 2D cross-section) so three.js can render pheromone trails and the painted tumor/wound region alongside agent positions.
